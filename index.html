function renderTree(){ const svg = document.getElementById("treeSvg"); const title = document.getElementById("activeTitle"); const subtitle = document.getElementById("activeSubtitle"); svg.innerHTML = ""; const tree = findTree(activeTreeId); if(!tree){ title.textContent = "Simple tracing tree"; subtitle.textContent = "Search, create, or report to show an actual tree here."; drawSampleTree(svg); return; } title.textContent = tree.title; subtitle.textContent = `Creator: ${tree.creator} · Reports: ${flattenNodes(tree.children).length} · Created: ${tree.createdAt}`; const layout = computeLayout(tree); const width = Math.max(760, layout.width + 120); const height = Math.max(390, layout.height + 100); svg.setAttribute("viewBox", `0 0 ${width} ${height}`); svg.setAttribute("height", height); /* Draw connector lines first, so vertices appear above the lines. These are small straight SVG line elements between parent and child. */ for(const edge of layout.edges){ const line = document.createElementNS("http://www.w3.org/2000/svg","line"); line.setAttribute("x1", edge.x1); line.setAttribute("y1", edge.y1); line.setAttribute("x2", edge.x2); line.setAttribute("y2", edge.y2); line.setAttribute("stroke", "#bfc7bb"); line.setAttribute("stroke-width", "2"); line.setAttribute("stroke-linecap", "round"); line.setAttribute("opacity", "0.9"); svg.appendChild(line); } for(const node of layout.nodes){ const circle = document.createElementNS("http://www.w3.org/2000/svg","circle"); circle.setAttribute("cx", node.x); circle.setAttribute("cy", node.y); circle.setAttribute("r", node.type === "root" ? 18 : 15); circle.setAttribute("class", `node-circle ${node.type === "root" ? "node-root" : "node-report"} ${node.id === activeNodeId ? "node-selected" : ""}`); circle.addEventListener("click", () => openNodePopup(node.id)); svg.appendChild(circle); const text = document.createElementNS("http://www.w3.org/2000/svg","text"); text.setAttribute("x", node.x); text.setAttribute("y", node.y + 34); text.setAttribute("class", "node-label"); text.textContent = shorten(node.label, 18); svg.appendChild(text); const voteText = document.createElementNS("http://www.w3.org/2000/svg","text"); voteText.setAttribute("x", node.x); voteText.setAttribute("y", node.y + 50); voteText.setAttribute("class", "node-label"); voteText.textContent = `votes: ${node.votes || 0}`; svg.appendChild(voteText); } }
