function renderTree(){ const svg = document.getElementById("treeSvg"); const title = document.getElementById("activeTitle"); const subtitle = document.getElementById("activeSubtitle"); svg.innerHTML = ""; const tree = findTree(activeTreeId); if(!tree){ title.textContent = "Simple tracing tree"; subtitle.textContent = "Search, create, or report to show an actual tree here."; drawSampleTree(svg); return; } title.textContent = tree.title; subtitle.textContent = `Creator: ${tree.creator} · Reports: ${flattenNodes(tree.children).length} · Created: ${tree.createdAt}`; const layout = computeLayout(tree); const width = Math.max(760, layout.width + 120); const height = Math.max(390, layout.height + 100); svg.setAttribute("viewBox", `0 0 ${width} ${height}`); svg.setAttribute("height", height); /* Draw relation lines first. They must be placed before vertices so circles stay visible above lines. */ for(const edge of layout.edges){ const line = document.createElementNS("http://www.w3.org/2000/svg","line"); line.setAttribute("x1", edge.x1); line.setAttribute("y1", edge.y1); line.setAttribute("x2", edge.x2); line.setAttribute("y2", edge.y2); line.setAttribute("stroke", "#bfc7bb"); line.setAttribute("stroke-width", "2"); line.setAttribute("stroke-linecap", "round"); line.setAttribute("opacity", "0.95"); svg.appendChild(line); } /* Draw vertices after the lines. */ for(const node of layout.nodes){ const circle = document.createElementNS("http://www.w3.org/2000/svg","circle"); circle.setAttribute("cx", node.x); circle.setAttribute("cy", node.y); circle.setAttribute("r", node.type === "root" ? 18 : 15); circle.setAttribute( "class", `node-circle ${node.type === "root" ? "node-root" : "node-report"} ${node.id === activeNodeId ? "node-selected" : ""}` ); circle.addEventListener("click", () => openNodePopup(node.id)); svg.appendChild(circle); const text = document.createElementNS("http://www.w3.org/2000/svg","text"); text.setAttribute("x", node.x); text.setAttribute("y", node.y + 34); text.setAttribute("class", "node-label"); text.textContent = shorten(node.label, 18); svg.appendChild(text); const voteText = document.createElementNS("http://www.w3.org/2000/svg","text"); voteText.setAttribute("x", node.x); voteText.setAttribute("y", node.y + 50); voteText.setAttribute("class", "node-label"); voteText.textContent = `votes: ${node.votes || 0}`; svg.appendChild(voteText); } } function computeLayout(tree){ const levelGap = 110; const leafGap = 95; let cursor = 70; const nodes = []; const edges = []; const nodeMap = new Map(); /* First pass: assign x/y positions to every node. */ function assignPosition(node, depth){ const children = node.children || []; let x; if(children.length === 0){ x = cursor; cursor += leafGap; }else{ const childXs = []; for(const child of children){ const childPosition = assignPosition(child, depth + 1); childXs.push(childPosition.x); } x = (Math.min(...childXs) + Math.max(...childXs)) / 2; } const y = 55 + depth * levelGap; const positioned = { id: node.id, label: node.title || node.suspect || "node", type: node.title ? "root" : "report", x, y, votes: node.votes || 0, raw: node }; nodes.push(positioned); nodeMap.set(node.id, positioned); return positioned; } /* Second pass: create parent-child edges after all coordinates are known. */ function assignEdges(parent){ const parentPosition = nodeMap.get(parent.id); for(const child of parent.children || []){ const childPosition = nodeMap.get(child.id); if(parentPosition && childPosition){ edges.push({ x1: parentPosition.x, y1: parentPosition.y, x2: childPosition.x, y2: childPosition.y }); } assignEdges(child); } } assignPosition(tree, 0); assignEdges(tree); const maxDepth = getDepth(tree); return { nodes, edges, width: cursor, height: 80 + maxDepth * levelGap }; }
